--Вывести Кол-во записей в справочнике товаров, удовлетворяющих следующим условиям:
--название товара содержит "томат" (без учета регистра) 
--и объем единицы товара больше 1-го литра.
SELECT COUNT(*) AS tomato_count FROM anp.t_sqlt_art
WHERE SUBSTRING(lower(NAME), 'томат') = 'томат'
AND VOL_TRANSP > 1
